<script>
  const identifyUrl    = "http://localhost:5678/webhook/identify";
  const confirmUrl     = "http://localhost:5678/webhook/confirm";
  const inventoryUrl   = "http://localhost:5678/webhook/inventory";
  const resetUrl       = "http://localhost:5678/webhook/reset";
  const rfidPendingUrl = "http://localhost:5678/webhook/rfid-pending";

  let requestId        = "";
  let detectedMedicine = "";
  let selectedTimes    = [];   // array of "HH:MM" strings
  let rfidPollingTimer = null;
  let detectedRFID     = "";
  let lastSeenUID      = "";
  let rfidWaitStart    = 0;

  /* ───── BADGE / MSG ───── */
  function setBadge(type, text) {
    const el = document.getElementById("statusBadge");
    el.className = "status-badge badge-" + type;
    document.getElementById("statusText").textContent = text;
  }
  function setMsg(html) {
    document.getElementById("message").innerHTML = html;
  }

  /* ───── STEPS ───── */
  function showStep(id) {
    ["step-upload","step-select","step-rfid","step-reupload"].forEach(s => {
      document.getElementById(s).classList.toggle("hidden", s !== id);
    });
  }

  /* ───── TIME PICKER ───── */
  function renderTimeTags() {
    const container = document.getElementById("timeTags");
    const empty     = document.getElementById("timeEmpty");
    // remove existing tags (not the empty placeholder)
    container.querySelectorAll(".time-tag").forEach(el => el.remove());

    if (selectedTimes.length === 0) {
      empty.style.display = "";
    } else {
      empty.style.display = "none";
      selectedTimes.forEach(t => {
        const tag = document.createElement("span");
        tag.className = "time-tag";
        tag.innerHTML = `${t} <button class="time-tag-remove" onclick="removeTime('${t}')" title="Remove">×</button>`;
        container.appendChild(tag);
      });
    }
  }

  function addTime() {
    const val = document.getElementById("timeInput").value;
    if (!val) return;
    if (selectedTimes.includes(val)) return; // no duplicates
    selectedTimes.push(val);
    selectedTimes.sort(); // keep times in chronological order
    renderTimeTags();
  }

  function removeTime(t) {
    selectedTimes = selectedTimes.filter(x => x !== t);
    renderTimeTags();
  }

  // Allow pressing Enter in the time input to add
  document.addEventListener("DOMContentLoaded", () => {
    document.getElementById("timeInput").addEventListener("keydown", e => {
      if (e.key === "Enter") addTime();
    });
  });

  /* ───── RESET UI ───── */
  function resetUI() {
    stopRFIDPolling();
    requestId = ""; detectedMedicine = ""; selectedTimes = []; detectedRFID = ""; lastSeenUID = "";
    document.getElementById("imageInput").value = "";
    document.getElementById("rfidUidDisplay").classList.add("hidden");
    document.getElementById("confirmBtn").classList.add("hidden");
    renderTimeTags();
    setBadge("info", "Ready");
    setMsg("Select a medicine image to begin identification.");
    showStep("step-upload");
  }

  /* ───── RESET SYSTEM ───── */
  async function resetSystem() {
    if (!confirm("Reset the entire system? All boxes will be cleared.")) return;
    await fetch(resetUrl, { method: "POST" });
    resetUI();
    loadInventory();
    alert("System reset complete.");
  }

  /* ───── UPLOAD IMAGE ───── */
  async function uploadImage() {
    const file = document.getElementById("imageInput").files[0];
    if (!file) return alert("Please select an image first.");
    setBadge("warn", "Identifying…");
    setMsg("Uploading image and identifying medicine…");

    const fd = new FormData();
    fd.append("image", file);
    fd.append("user", "demo_user");

    try {
      const res  = await fetch(identifyUrl, { method:"POST", body:fd });
      const data = await res.json();

      requestId        = data.request_id;
      detectedMedicine = data.medicine || "Unknown";

      setBadge("info", "Review");
      setMsg(`Detected medicine: <b>${detectedMedicine}</b><br>Choose a box and set dispensing times, then confirm.`);

      const boxSel = document.getElementById("boxSelect");
      boxSel.innerHTML = "";
      (data.free_boxes || []).forEach(b => {
        boxSel.innerHTML += `<option value="${b}">Box ${b}</option>`;
      });

      // Reset time picker for new session
      selectedTimes = [];
      renderTimeTags();
      document.getElementById("timeInput").value = "08:00";

      showStep("step-select");

    } catch(e) {
      setBadge("warn", "Error");
      setMsg("Could not reach the server. Check n8n is running.");
      showStep("step-upload");
    }
  }

  /* ───── PROCEED TO RFID ───── */
  async function proceedToRFID() {
    if (selectedTimes.length === 0) return alert("Add at least one dispensing time.");

    const box = document.getElementById("boxSelect").value;
    setBadge("warn", "RFID Pending");
    setMsg(`Medicine: <b>${detectedMedicine}</b> · Box: <b>${box}</b><br>Waiting for RFID card tap…`);
    document.getElementById("rfidUidDisplay").classList.add("hidden");
    document.getElementById("confirmBtn").classList.add("hidden");
    showStep("step-rfid");

    lastSeenUID  = "";
    detectedRFID = "";

    // Cancel any pending auto-reset from a previous completed session
    if (window._resetTimer) { clearTimeout(window._resetTimer); window._resetTimer = null; }

    // Fire-and-forget clear — don't await, start polling after short delay
    fetch("http://localhost:5678/webhook/rfid-clear", { method: "POST" })
      .catch(() => {});

    rfidWaitStart = Date.now();
    setTimeout(startRFIDPolling, 1500);
  }

  function cancelRFID() {
    stopRFIDPolling();
    setBadge("info", "Review");
    setMsg(`Detected medicine: <b>${detectedMedicine}</b><br>Adjust box and times, then confirm.`);
    showStep("step-select");
  }

  /* ───── RFID POLLING ───── */
  let rfidPollInFlight = false;

  function startRFIDPolling() {
    stopRFIDPolling();
    rfidPollInFlight = false;
    rfidPollingTimer = setInterval(pollForRFID, 2000);
  }

  function stopRFIDPolling() {
    if (rfidPollingTimer) { clearInterval(rfidPollingTimer); rfidPollingTimer = null; }
    rfidPollInFlight = false;
  }

  async function pollForRFID() {
    if (rfidPollInFlight || !rfidPollingTimer) return;

    try {
      rfidPollInFlight = true;
      const res  = await fetch(rfidPendingUrl);
      const body = await res.json();
      rfidPollInFlight = false;

      const obj  = Array.isArray(body) ? (body[0] || {}) : body;
      const uid  = (obj.uid || "").trim();

      console.log("Poll rfid-pending:", uid, "| lastSeen:", lastSeenUID);

      if (!rfidPollingTimer) return;  // polling was stopped while fetch was in flight

      if (uid !== "" && uid !== lastSeenUID) {
        stopRFIDPolling();
        lastSeenUID  = uid;
        detectedRFID = uid;

        document.getElementById("rfidUidValue").textContent = detectedRFID;
        document.getElementById("rfidUidDisplay").classList.remove("hidden");
        document.getElementById("confirmBtn").classList.remove("hidden");

        setBadge("success", "RFID Verified");
        setMsg(`✅ Card <span class="highlight">${detectedRFID}</span> verified. Click Save &amp; Confirm to complete.`);
      }
    } catch(e) {
      rfidPollInFlight = false;
      console.warn("pollForRFID error:", e);
    }
  }

  /* ───── FINAL CONFIRM ───── */
  async function finalConfirm() {
    const box = document.getElementById("boxSelect").value;
    setBadge("warn", "Saving…");
    setMsg("Storing medicine assignment…");
    document.getElementById("confirmBtn").disabled = true;

    try {
      await fetch(confirmUrl, {
        method: "POST",
        headers: { "Content-Type":"application/json" },
        body: JSON.stringify({
          request_id:     requestId,
          selected_box:   box,
          selected_times: selectedTimes,
          rfid_id:        detectedRFID,
          medicine:       detectedMedicine
        })
      });

      setBadge("success", "Saved");
      setMsg(`✅ <b>${detectedMedicine}</b> assigned to Box <b>${box}</b> and RFID recorded.`);
      loadInventory();
      if (window._resetTimer) clearTimeout(window._resetTimer);
      window._resetTimer = setTimeout(() => { window._resetTimer = null; resetUI(); }, 5000);

    } catch(e) {
      setBadge("warn", "Error");
      setMsg("Save failed. Please try again.");
      document.getElementById("confirmBtn").disabled = false;
    }
  }

  /* ───── REJECT ───── */
  function rejectSelection() {
    setBadge("info", "Try Again");
    setMsg("Medicine not correct. Please re-upload a clearer image.");
    showStep("step-reupload");
  }

  /* ───── INVENTORY ───── */
  async function loadInventory() {
    try {
      const res  = await fetch(inventoryUrl);
      const rows = await res.json();
      const div  = document.getElementById("inventoryList");
      div.innerHTML = "";
      for (let b = 1; b <= 5; b++) {
        const item   = rows.find(r => Number(r.box) === b);
        const filled = item && item.medicine;
        let times = "";
        if (item && item.times) {
          try { times = JSON.parse(item.times).join(" · "); } catch { times = item.times; }
        }
        div.innerHTML += `
          <div class="inv-box ${filled ? "filled" : ""}">
            <div class="inv-box-num">${b}</div>
            <div class="inv-box-name">${filled ? item.medicine : "—"}</div>
            ${times ? `<div class="inv-box-times">${times}</div>` : ""}
          </div>`;
      }
    } catch(e) { /* silent */ }
  }

  /* ───── INIT ───── */
  renderTimeTags();
  loadInventory();
  setInterval(loadInventory, 5000);
</script>
